```js
export function dayKeyJST(ms) {
  return new Date(ms + JST_OFFSET_MS).toISOString().slice(0, 10);
}

export function jstMidnightMs(ms) {
  return Date.parse(dayKeyJST(ms) + "T00:00:00+09:00");
}

export function parseUsageLine(line) {
  if (!line.includes('"usage"')) return null;
  let o;
  try { o = JSON.parse(line); } catch { return null; }
  const u = o?.message?.usage;
  if (!u || !o.timestamp) return null;
  const ts = Date.parse(o.timestamp);
  if (Number.isNaN(ts)) return null;
  const tokens =
    (u.input_tokens || 0) + (u.output_tokens || 0) +
    (u.cache_creation_input_tokens || 0) + (u.cache_read_input_tokens || 0);
  const id = o.message.id || "";
  const req = o.requestId || "";
  return { key: id || req ? `${id}:${req}` : null, day: dayKeyJST(ts), tokens };
}
```

```js
export function stageFor(total) {
  let i = THRESHOLDS.length - 1;
  while (i > 0 && total < THRESHOLDS[i]) i--;
  const floor = THRESHOLDS[i];
  const ceil = i + 1 < THRESHOLDS.length ? THRESHOLDS[i + 1] : null;
  return { index: i, floor, ceil, progress: ceil === null ? 1 : (total - floor) / (ceil - floor) };
}

export function computePace(samples) {
  if (samples.length < 2) return null;
  const first = samples[0];
  const last = samples[samples.length - 1];
  const span = last.t - first.t;
  if (span < 5 * 60e3) return null;
  return Math.max(0, ((last.total - first.total) / span) * 3600e3);
}
```

```js
export function strWidth(s) {
  let w = 0;
  for (const ch of s) {
    const c = ch.codePointAt(0);
    const wide =
      (c >= 0x1100 && c <= 0x115f) ||
      (c >= 0x2e80 && c <= 0xa4cf) ||
      (c >= 0xac00 && c <= 0xd7a3) ||
      (c >= 0xf900 && c <= 0xfaff) ||
      (c >= 0xff00 && c <= 0xff60) ||
      (c >= 0xffe0 && c <= 0xffe6) ||
      (c >= 0x1f300 && c <= 0x1faff);
    w += wide ? 2 : 1;
  }
  return w;
}

export function makeBubble(text) {
  const w = strWidth(text);
  return [
    "╭" + "─".repeat(w + 2) + "╮",
    "│ " + text + " │",
    "╰" + "─".repeat(w + 2) + "╯",
  ];
}

export function gauge(progress, width) {
  const p = Math.min(1, Math.max(0, progress));
  const filled = Math.round(p * width);
  return "█".repeat(filled) + "░".repeat(width - filled);
}

export function formatPace(pace) {
  if (pace === null) return "計測中…";
  if (pace >= 1e6) return (pace / 1e6).toFixed(1) + "M tokens/h";
  return Math.round(pace / 1e3) + "K tokens/h";
}
```

```js
export function createCollector(root, now = Date.now) {
  const files = new Map(); // path -> { offset, remainder }
  const daily = new Map(); // "YYYY-MM-DD" -> total
  const seen = new Set();  // "messageId:requestId"
  let seenDay = dayKeyJST(now());

  function rollover(today) {
    seen.clear();
    for (const k of [...daily.keys()]) if (k !== today) daily.delete(k);
    seenDay = today;
  }

  function ingestLine(line) {
    const p = parseUsageLine(line);
    if (!p) return;
    if (p.key) {
      if (seen.has(p.key)) return;
      seen.add(p.key);
    }
    daily.set(p.day, (daily.get(p.day) || 0) + p.tokens);
  }

  function processFile(fp, st, midnight) {
    let rec = files.get(fp);
    if (!rec) {
      // 初見のファイル。今日書き込みがないものは中身を読まずスキップ位置だけ覚える
      rec = { offset: st.mtimeMs < midnight ? st.size : 0, remainder: "" };
      files.set(fp, rec);
    }
    if (st.size < rec.offset) { rec.offset = 0; rec.remainder = ""; } // 縮小 → 読み直し
    if (st.size === rec.offset) return;
    const fd = fs.openSync(fp, "r");
    try {
      const buf = Buffer.alloc(st.size - rec.offset);
      fs.readSync(fd, buf, 0, buf.length, rec.offset);
      rec.offset = st.size;
      const lines = (rec.remainder + buf.toString("utf8")).split("\n");
      rec.remainder = lines.pop(); // 書き込み途中の最終行は次回へ持ち越し
      for (const line of lines) ingestLine(line);
    } finally {
      fs.closeSync(fd);
    }
  }

  function scan() {
    const t = now();
    const today = dayKeyJST(t);
    if (today !== seenDay) rollover(today);
    const midnight = jstMidnightMs(t);
    const stack = [root];
    while (stack.length) {
      const dir = stack.pop();
      let entries;
      try { entries = fs.readdirSync(dir, { withFileTypes: true }); } catch { continue; }
      for (const e of entries) {
        const fp = path.join(dir, e.name);
        if (e.isDirectory()) {
          stack.push(fp);
        } else if (e.isFile() && e.name.endsWith(".jsonl")) {
          let st;
          try { st = fs.statSync(fp); } catch { continue; }
          try { processFile(fp, st, midnight); } catch { /* 次の tick で再試行 */ }
        }
      }
    }
  }

  return {
    scan,
    todayTotal: () => daily.get(dayKeyJST(now())) || 0,
  };
}
```

```js
export const ART_H = 11;       // 最大アート高（Stage 6 の 10 行 + 余白 1）
const ART_LEFT = 4;            // アートの左マージン
const BUBBLE_COL = 28;         // 吹き出しの開始カラム
const GAUGE_W = 22;

export function composeScreen(view) {
  const lines = [];
  const art = view.artLines;
  const pad = ART_H - art.length; // 下揃え
  const bubble = makeBubble(view.bubbleText);
  for (let i = 0; i < ART_H; i++) {
    let line = i >= pad ? " ".repeat(ART_LEFT) + art[i - pad] : "";
    if (i >= 1 && i - 1 < bubble.length) {
      line += " ".repeat(Math.max(1, BUBBLE_COL - strWidth(line))) + bubble[i - 1];
    }
    lines.push(line);
  }
  lines.push("");
  lines.push(`  Stage ${view.stageIndex + 1} ── ${view.stageName}`);
  lines.push(`  Today : ${view.total.toLocaleString("en-US")} tokens`);
  if (view.ceil === null) {
    lines.push(`  [${gauge(1, GAUGE_W)}] MAX`);
  } else {
    lines.push(`  [${gauge(view.progress, GAUGE_W)}] ${Math.floor(view.progress * 100)}% → Stage ${view.stageIndex + 2}`);
  }
  lines.push(`  Pace  : ${formatPace(view.pace)}`);
  return lines;
}
```

```js
function runLoop() {
  const color = detectColor();
  const collector = createCollector(PROJECTS_ROOT);
  let samples = [];
  let lastDay = dayKeyJST(Date.now());
  let speech = "きょうも いっしょに がんばろ";
  let lastSpeechAt = 0;
  let frameIdx = 0;
  let frameToggleAt = 0;
  let blinkUntil = 0;
  let nextBlinkAt = Date.now() + 3000 + Math.random() * 5000;

  const out = process.stdout;
  out.write("\x1b[?1049h\x1b[?25l"); // 代替スクリーン + カーソル非表示
  let cleaned = false;
  const cleanup = () => {
    if (cleaned) return;
    cleaned = true;
    out.write("\x1b[?25h\x1b[?1049l");
  };
  process.on("SIGINT", () => { cleanup(); process.exit(0); });
  process.on("SIGTERM", () => { cleanup(); process.exit(0); });
  process.on("exit", cleanup);

  const dataTick = () => {
    const now = Date.now();
    const today = dayKeyJST(now);
    if (today !== lastDay) { samples = []; lastDay = today; } // 日付またぎでペースをリセット
    collector.scan();
    samples.push({ t: now, total: collector.todayTotal() });
    samples = samples.filter((s) => now - s.t <= 3600e3); // 直近 1 時間分だけ保持
  };

  const renderTick = () => {
    const now = Date.now();
    const total = collector.todayTotal();
    const s = stageFor(total);
    const stage = STAGES[s.index];

    if (now >= frameToggleAt) { frameIdx++; frameToggleAt = now + 1000; }
    if (now >= nextBlinkAt) { blinkUntil = now + 400; nextBlinkAt = now + 4000 + Math.random() * 6000; }
    const art = stage.blink && now < blinkUntil ? stage.blink : stage.frames[frameIdx % stage.frames.length];

    if (now - lastSpeechAt >= SPEECH_INTERVAL_MS) {
      speech = pickSpeech({ stageIndex: s.index, pace: computePace(samples), hour: hourJST(now), total });
      lastSpeechAt = now;
    }

    const cols = out.columns || 80;
    const rows = out.rows || 24;
    if (cols < 60 || rows < ART_H + 6) {
      out.write("\x1b[H\x1b[2J  ウィンドウを ひろげてね\n");
      return;
    }

    const lines = composeScreen({
      artLines: art, bubbleText: speech, stageIndex: s.index, stageName: stage.name,
      total, progress: s.progress, ceil: s.ceil, pace: computePace(samples),
    });
    let buf = "\x1b[H";
    lines.forEach((line, i) => {
      const colored = i < ART_H ? color + line + RESET : line; // ペットと吹き出しはオレンジ
      buf += colored + "\x1b[K\n";
    });
    buf += "\x1b[J";
    out.write(buf);
  };

  dataTick();
  renderTick();
  setInterval(dataTick, DATA_INTERVAL_MS);
  setInterval(renderTick, RENDER_INTERVAL_MS);
}

function main() {
  const args = process.argv.slice(2);
  if (args.includes("--preview")) return preview();
  if (args.includes("--once")) return once();
  runLoop();
}
```
